二叉搜索树的比较判别

题目描述：

对于输入的各种序列，判断是否能生成一样的二叉搜索树：

输入样例：

代码描述：

#include<stdio.h>
#include<stdlib.h>
typedef struct tree * BinTree;
struct tree{
    int data;
    int flag;
    BinTree left;
    BinTree right;
};
int CheckTree(BinTree root,int data);
void PrintTree(BinTree BST)
{
    if(!BST) return;
    else
    {
        printf("%d",BST->data);
        PrintTree(BST->left);
        PrintTree(BST->right);
    }    
}
BinTree NewNode(int data)
{
    BinTree BST = (BinTree) malloc(sizeof(struct tree));
    BST->data = data;
    BST->flag = 0;
    BST->left = NULL;
    BST->right = NULL;
    return BST;
}
BinTree InsertTree(BinTree BST,int data)
{
    if(!BST) 
    {
        BST = NewNode(data);
    }
    else
    {
        if(data>BST->data)
        {
            BST->right = InsertTree(BST->right,data);
        }
        if(data<BST->data)
        {
            BST->left = InsertTree(BST->left,data);
        }
    }
    return BST;
}
BinTree MakeTree(int n)
{
    int i;
    int data;
    BinTree root = NULL;
    for(i=0;i<n;i++)
    {
        scanf("%d",&data);
        root = InsertTree(root,data);
    }
    return root;
}
int JudgeTree(int n,BinTree root)
{
    int i;
    int data;
    int flag = 1;

    scanf("%d",&data);

    if(data==root->data)
    {
        root->flag = 1;
    }
    else
    {
        flag = 0;
    }

    for(i=0;i<n-1;i++)
    {
        scanf("%d",&data);

        if(flag) //如果flag为0，便不再需要判断了
        {
            flag = CheckTree(root,data);//每次查找都是从根节点开始查找
        }    
    }
    if(flag) return 1;
    else return 0;
}
int CheckTree(BinTree root,int data)
{            
    if(root->flag==1)
    {
        if(data>root->data)
        {
            return CheckTree(root->right,data);
        }
        else if(data<root->data)
        {
            return CheckTree(root->left,data);
        }
        else return 0;//是不存在一个数字出现两次的情况
    }
    else
    {
        if(data==root->data)//这是他应该放的位置，如果数据不一样就必然错误
        {
            root->flag = 1;
            return 1;
        }
        else return 0;    
    }
}
void CleanFlag(BinTree root)
{
    if(!root)
    {
        root->flag = 0;
        CleanFlag(root->right);
        CleanFlag(root->left);
    }
}
void FreeTree(BinTree root)
{
    if(!root)
    {
        FreeTree(root->left);
        FreeTree(root->right);
        free(root);
    }
}
int main()
{
    int N;
    int L;
    int i;
    BinTree root = NULL;
    scanf("%d",&N);
while(N)
{
    scanf("%d",&L);

    root = MakeTree(N);    
    //PrintTree(root);
    for(i=0;i<L;i++)
    {
        if(JudgeTree(N,root))
        {
            printf("yes\n");
        }
        else
        {
            printf("no\n");
        }
        CleanFlag(root);//判别之后需要将树中的flag位置置零
    }

    FreeTree(root);//结束一棵树的多次判断之后需要将树清空

    scanf("%d",&N);
}
return 0;
}

代码个人理解:

解决此种问题的关键在于 CheckTree(root,data) 函数，这个函数的功能在于给他一个根节点和一个数，他来检索出这个数是不是在以这个根节点领出来的树中一样的位置。
（输入的数字所在的位置是根据之前一连串的数字建树形成，只是这边并没有切实的建树）
此函数执行的过程实际上就是将数字再放入一遍树，当遇到root->flag为0的时候就是这个数字应该摆放的位置，此时只要比较树中节点的值与传进来的数值是否一样便可。
