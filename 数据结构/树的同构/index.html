树的同构

发表于 2020-02-13 分类于 数据结构

#题意：
给定两棵树T1和T2，如果T1可以通过若干次左右孩子互换就变成T2，则我们称两棵树是“同构”的。现给定两棵树，请你判断它们是否是同构的。

输入样例如下图：

求解思路：

一、二叉树的表示：

这一题中我用的是结构体数组来表示的，结构体中包含节点数值与左右孩子编号，如果没有孩子则编号为-1。
以下是我定义的数据结构：

struct tree{
    int data;
    int left;
    int right
} Tree1[MAX],Tree2[MAX];

二、建立二叉树：

我用了BuildTree()来进行建树，建树过程较为简单，为简单的循环输入。

三、同构判别：

在写同构判别的函数的时候我用了递归的思想然后分多种情况讨论。

源码：

#include<stdio.h>

#define MAX 10
#define Null -1

int HashTable[MAX];

struct tree{
    int data;
    int left;
    int right
} Tree1[MAX],Tree2[MAX];

void BuildTree(struct tree * Tree,int n)
{    
int i = 0;

while(i<n)
{
    scanf("%d%d%d",&Tree[i].data,&Tree[i].left,&Tree[i].right);    
    i++;
}
}

int FindRoot(struct tree * Tree,int n)
{
int i;
int ans = -1;

for(i=0;i<n;i++)
{
    if(Tree[i].left!=-1)
    {
        HashTable[Tree[i].left]++;
    }
    if(Tree[i].right!=-1)
    {
        HashTable[Tree[i].right]++;
    }
}

for(i=0;i<n;i++)
{
    if(HashTable[i]==0)
    {
        ans = i;
        break;
    }
}
for(i=0;i<n;i++)
{
    HashTable[i]=0;
}
return ans;
}

int JudgeTree(int root1,int root2)
{
if((root1==Null)&&(root2==Null))//如果根节点都空，则必然为同构 
return 1;

else if((root1==Null)&&(root2!=Null))//如果根节点一个空一个不空，则必然不同构 
return 0;

else if((root1!=Null)&&(root2==Null))//如果根节点一个空一个不空，则必然不同构 
return 0;

else if(Tree1[root1].data!=Tree2[root2].data)//如果根节点的数据值不一样则必然不同构 
return 0;

else if((Tree1[root1].left==Null)&&(Tree2[root2].left==Null))//如果左边都空，就看看右边是不是同构的 
return JudgeTree(Tree1[root1].right,Tree2[root2].right);

else if((Tree1[root1].right==Null)&&(Tree2[root2].right==Null))//如果右边都空，就看看左边是不是同构的 
return JudgeTree(Tree1[root1].left,Tree2[root2].left);

else if((Tree1[root1].right!=Null)&&(Tree2[root2].right==Null))//一个空一个不空，则看看都不空的那一边是否同构 
return JudgeTree(Tree1[root1].right,Tree2[root2].left);

else if((Tree1[root1].right==Null)&&(Tree2[root2].right!=Null))//一个空一个不空，则看看都不空的那一边是否同构 
return JudgeTree(Tree1[root1].left,Tree2[root2].right);

else if((Tree1[root1].right==Null)&&(Tree1[root1].left==Null)&&(Tree2[root2].right==Null)&&(Tree2[root2].left==Null))//如果两边都空，就看看根节点数据是否一样 
{
     if(Tree1[root1].data!=Tree2[root2].data)
     return 0;
     else return 1; 
} 

else if((Tree1[root1].right!=Null)&&(Tree1[root1].left!=Null)&&(Tree2[root2].right!=Null)&&(Tree2[root2].left!=Null))//如果都不空 
{
    if(Tree1[Tree1[root1].left].data==Tree2[Tree2[root2].left].data)// 左边的数值一样的话，就只看左边和左边是不是同构 
    return ( JudgeTree(Tree1[root1].left,Tree2[root2].left)&&JudgeTree(Tree1[root1].right,Tree2[root2].right) );
    else return ( JudgeTree(Tree1[root1].left,Tree2[root2].right)&&JudgeTree(Tree1[root1].right,Tree2[root2].left) ); 
}        
}

int main()
{
    int n;
    int root1,root2;
    int ans;

scanf("%d",&n);

BuildTree(Tree1,n);
BuildTree(Tree2,n);

root1 = FindRoot(Tree1,n);
root2 = FindRoot(Tree2,n);

ans = JudgeTree(root1,root2);

printf("%d",ans);


return 0;
}

注：在使用递归函数的时候，我们需要先求出二叉树的根节点，因为JudgeTree()这个函数的作用是，给他根节点，他来判断树是不是同构的，在函数下面递归调用的时候传进去的参数分别是左右子树的根节点，他的含义同样是判断传进来树是不是同构。（他是通过传进来的根节点来判断要判断的是哪两棵树）。
